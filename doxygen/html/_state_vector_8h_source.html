347  bool remove( const T& value, ActionSource source = ActionSource::NONE_E )
348  {
349  // NOTE: State variables can only be set from the application thread
350  ASSERT_IS_APPLICATION_THREAD_OR_INITIALIZING();
351 
352  bool removed = false;
353  {
354  StateEngine::lock_type lock( StateEngine::GetMutex() );
355  typename value_type::iterator it = std::find( this->values_vector_.begin(),
356  this->values_vector_.end(), value );
357  if ( it != this->values_vector_.end() )
358  {
359  this->values_vector_.erase( it );
360  removed = true;
361  }
362  }
363 
364  if ( removed && this->signals_enabled() )
365  {
366  this->value_changed_signal_( this->values_vector_, source );
367  this->state_changed_signal_();
368  }
369  return removed;
