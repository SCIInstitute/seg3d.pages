245 template<class T>
246 bool ITKImage2DDataT<T>::initialize( DataBlockHandle data_block,
247  Transform transform, SliceType slice )
248 {
249  // Step (1) : convert image data to the right type
250  // NOTE: This conversion is done on the fly here
251 
252  DataType image_data_type = GetDataType( reinterpret_cast<T*>(0) );
253  if ( data_block->get_data_type() != image_data_type )
254  {
255  if ( !( DataBlock::ConvertDataType( data_block, data_block_, image_data_type ) ) )
256  {
257  data_block_.reset();
258  }
259  }
260  else
261  {
262  data_block_ = data_block;
263  }
264 
265  // If the data block could not be generated bail out of the code
266  if ( !data_block_ ) return false;
267 
268  // Step (2) : Use itk factory method for building new itk image object
269  itk_image_ = image_type::New();
270 
271  // Step (3) : copy the dimensions of the data block
272  typename image_type::RegionType region;
273 
274  switch ( slice )
275  {
276  case SliceType::SAGITTAL_E:
277  region.SetSize( 0, data_block->get_ny() );
278  region.SetSize( 1, data_block->get_nz() );
279  break;
280  case SliceType::CORONAL_E:
281  region.SetSize( 0, data_block->get_nx() );
282  region.SetSize( 1, data_block->get_nz() );
283  break;
284  case SliceType::AXIAL_E:
285  region.SetSize( 0, data_block->get_nx() );
286  region.SetSize( 1, data_block->get_ny() );
287  break;
288  }
289  itk_image_->SetRegions( region );
290 
291  // Step (4) : Wrap our data into an ITK pixel container
292  typename image_type::PixelContainerPointer pixel_container =
293  image_type::PixelContainer::New();
294 
295  pixel_container->SetImportPointer( reinterpret_cast<T*>( data_block->get_data() ),
296  static_cast<typename image_type::PixelContainer::ElementIdentifier >
297  ( data_block->get_size() ), false );
298 
299  itk_image_->SetPixelContainer( pixel_container );
300 
301  // Step (5) : Add the transformation to the itk data
302  set_transform( transform, slice );
303 
304  return true;
305 }

339 template<class T>
340 Transform ITKImage2DDataT<T>::get_transform() const
341 {
342  typename image_type::PointType point = itk_image_->GetOrigin();
343  Point origin( point[ 0 ], point[ 1 ], 0.0 );
344 
345  typename image_type::DirectionType direction = itk_image_->GetDirection();
346  typename image_type::SpacingType spacing = itk_image_->GetSpacing();
347  Vector direction_x( direction[ 0 ][ 0 ], direction[ 0 ][ 1 ], 0.0 );
348  Vector direction_y( direction[ 1 ][ 0 ], direction[ 1 ][ 1 ], 0.0 );
349  Vector direction_z( 0.0, 0.0, 1.0 );
350 
351  direction_x *= spacing[ 0 ];
352  direction_y *= spacing[ 1 ];
353 
354  return Transform( origin, direction_x, direction_y, direction_z );
355 }

357 template<class T>
358 void ITKImage2DDataT<T>::set_transform( Transform& transform, SliceType slice )
359 {
360  switch ( slice )
361  {
362  case SliceType::SAGITTAL_E :
363  {
364  Point origin = transform.project( Point( 0.0, 0.0, 0.0 ) );
365  Vector direction_x = transform.project( Vector( 0.0, 1.0, 0.0 ) );
366  Vector direction_y = transform.project( Vector( 0.0, 0.0, 1.0 ) );
367 
368  double spacing_x = direction_x.length();
369  double spacing_y = direction_y.length();
370 
371  direction_x.normalize();
372  direction_y.normalize();
373 
374  typename image_type::PointType point;
375  point[ 0 ] = origin[ 1 ];
376  point[ 1 ] = origin[ 2 ];
377 
378  itk_image_->SetOrigin( point );
379 
380  typename image_type::DirectionType direction;
381 
382  direction[ 0 ][ 0 ] = direction_x[ 1 ];
383  direction[ 0 ][ 1 ] = direction_x[ 2 ];
384 
385  direction[ 1 ][ 0 ] = direction_y[ 1 ];
386  direction[ 1 ][ 1 ] = direction_y[ 2 ];
387 
388  itk_image_->SetDirection( direction );
389 
390  typename image_type::SpacingType spacing;
391  spacing[ 0 ] = spacing_x;
392  spacing[ 1 ] = spacing_y;
393  itk_image_->SetSpacing( spacing );
394  }
395  break;
396 
397  case SliceType::CORONAL_E :
398  {
399  Point origin = transform.project( Point( 0.0, 0.0, 0.0 ) );
400  Vector direction_x = transform.project( Vector( 1.0, 0.0, 0.0 ) );
401  Vector direction_y = transform.project( Vector( 0.0, 0.0, 1.0 ) );
402 
403  double spacing_x = direction_x.length();
404  double spacing_y = direction_y.length();
405 
406  direction_x.normalize();
407  direction_y.normalize();
408 
409  typename image_type::PointType point;
410  point[ 0 ] = origin[ 0 ];
411  point[ 1 ] = origin[ 2 ];
412 
413  itk_image_->SetOrigin( point );
414 
415  typename image_type::DirectionType direction;
416 
417  direction[ 0 ][ 0 ] = direction_x[ 0 ];
418  direction[ 0 ][ 1 ] = direction_x[ 2 ];
419 
420  direction[ 1 ][ 0 ] = direction_y[ 0 ];
421  direction[ 1 ][ 1 ] = direction_y[ 2 ];
422 
423  itk_image_->SetDirection( direction );
424 
425  typename image_type::SpacingType spacing;
426  spacing[ 0 ] = spacing_x;
427  spacing[ 1 ] = spacing_y;
428  itk_image_->SetSpacing( spacing );
429  }
430  break;
431 
432  case SliceType::AXIAL_E :
433  {
434  Point origin = transform.project( Point( 0.0, 0.0, 0.0 ) );
435  Vector direction_x = transform.project( Vector( 1.0, 0.0, 0.0 ) );
436  Vector direction_y = transform.project( Vector( 0.0, 1.0, 0.0 ) );
437 
438  double spacing_x = direction_x.length();
439  double spacing_y = direction_y.length();
440 
441  direction_x.normalize();
442  direction_y.normalize();
443 
444  typename image_type::PointType point;
445  point[ 0 ] = origin[ 0 ];
446  point[ 1 ] = origin[ 1 ];
447 
448  itk_image_->SetOrigin( point );
449 
450  typename image_type::DirectionType direction;
451 
452  direction[ 0 ][ 0 ] = direction_x[ 0 ];
453  direction[ 0 ][ 1 ] = direction_x[ 1 ];
454 
455  direction[ 1 ][ 0 ] = direction_y[ 0 ];
456  direction[ 1 ][ 1 ] = direction_y[ 1 ];
457 
458  itk_image_->SetDirection( direction );
459 
460  typename image_type::SpacingType spacing;
461  spacing[ 0 ] = spacing_x;
462  spacing[ 1 ] = spacing_y;
463  itk_image_->SetSpacing( spacing );
464  }
465  break;
466  }
