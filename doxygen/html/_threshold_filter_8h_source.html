55  template< class T >
56  void threshold_data( Core::DataBlockHandle dst, double min_val, double max_val )
57  {
58  Core::DataBlockHandle src_data_block = this->src_layer_->
59  get_data_volume()->get_data_block();
60  T* src_data = reinterpret_cast< T* >( src_data_block->get_data() );
61  unsigned char* dst_data = reinterpret_cast< unsigned char* >( dst->get_data() );
62  size_t z_plane_size = src_data_block->get_nx() * src_data_block->get_ny();
63  size_t nz = src_data_block->get_nz();
64  size_t tenth_nz = nz / 10;
65 
66  // Lock the source data block
67  Core::DataBlock::shared_lock_type lock( src_data_block->get_mutex() );
68  size_t index = 0;
69  for ( size_t z = 0; z < nz; ++z )
70  {
71  for ( size_t i = 0; i < z_plane_size; ++i, ++index )
72  {
73  dst_data[ index ] = ( src_data[ index ] >= min_val &&
74  src_data[ index ] <= max_val ) ? 1 : 0;
75  }
76  if ( this->check_abort() )
77  {
78  return;
79  }
80  if ( tenth_nz > 0 && z > 0 && z % tenth_nz == 0 )
81  {
82  this->dst_layer_->update_progress_signal_( ( z * 0.5 ) / nz );
83  }
84  }
85  this->dst_layer_->update_progress_signal_( 0.5 );
86  }
