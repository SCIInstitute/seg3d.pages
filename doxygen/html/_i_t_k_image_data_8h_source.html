255 template<class T>
256 bool ITKImageDataT<T>::initialize( DataBlockHandle data_block, Transform transform )
257 {
258  // Step (1) : convert image data to the right type
259  // NOTE: This conversion is done on the fly here
260 
261  DataType image_data_type = GetDataType( reinterpret_cast<T*>(0) );
262  if ( data_block->get_data_type() != image_data_type )
263  {
264  if ( ! DataBlock::ConvertDataType( data_block, data_block_, image_data_type ) )
265  {
266  data_block_.reset();
267  }
268  }
269  else
270  {
271  data_block_ = data_block;
272  }
273 
274  // If the data block could not be generated bail out of the code
275  if ( ! data_block_ ) return false;
276 
277  // Step (2) : Use itk factory method for building new itk image object
278  itk_image_ = image_type::New();
279 
280  // Step (3) : copy the dimensions of the data block
281  typename image_type::RegionType region;
282  region.SetSize( 0, data_block->get_nx() );
283  region.SetSize( 1, data_block->get_ny() );
284  region.SetSize( 2, data_block->get_nz() );
285  itk_image_->SetRegions( region );
286 
287  // Step (4) : Wrap our data into an ITK pixel container
288  typename image_type::PixelContainerPointer pixel_container =
289  image_type::PixelContainer::New();
290 
291  pixel_container->SetImportPointer( reinterpret_cast<T*>( data_block->get_data() ),
292  static_cast<typename image_type::PixelContainer::ElementIdentifier >
293  ( data_block->get_size() ), false );
294 
295  itk_image_->SetPixelContainer( pixel_container );
296 
297  // Step (5) : Add the transformation to the itk data
298  set_transform( transform );
299 
300  return true;
301 }

337 template<class T>
338 Transform ITKImageDataT<T>::get_transform() const
339 {
340  typename image_type::PointType point = itk_image_->GetOrigin();
341  Point origin( point[ 0 ], point[ 1 ], point[ 2 ] );
342 
343  // ITK direction type is matrix of direction cosines
344  // Seg3D uses same LPS (Left, Posterior, Superior) 3D basis
345  typename image_type::DirectionType direction = itk_image_->GetDirection();
346  typename image_type::SpacingType spacing = itk_image_->GetSpacing();
347  Vector direction_x( direction[ 0 ][ 0 ], direction[ 0 ][ 1 ], direction[ 0 ][ 2 ] );
348  Vector direction_y( direction[ 1 ][ 0 ], direction[ 1 ][ 1 ], direction[ 1 ][ 2 ] );
349  Vector direction_z( direction[ 2 ][ 0 ], direction[ 2 ][ 1 ], direction[ 2 ][ 2 ] );
350 
351  direction_x *= spacing[ 0 ];
352  direction_y *= spacing[ 1 ];
353  direction_z *= spacing[ 2 ];
354 
355  return Transform( origin, direction_x, direction_y, direction_z );
356 }

358 template<class T>
359 void ITKImageDataT<T>::set_transform( Transform& transform )
360 {
361  Point origin = transform.project( GridTransform::DEFAULT_ORIGIN );
362  Vector direction_x = transform.project( GridTransform::X_AXIS );
363  Vector direction_y = transform.project( GridTransform::Y_AXIS );
364  Vector direction_z = transform.project( GridTransform::Z_AXIS );
365 
366  double spacing_x = direction_x.length();
367  double spacing_y = direction_y.length();
368  double spacing_z = direction_z.length();
369 
370  direction_x.normalize();
371  direction_y.normalize();
372  direction_z.normalize();
373 
374  typename image_type::PointType point;
375  point[ 0 ] = origin[ 0 ];
376  point[ 1 ] = origin[ 1 ];
377  point[ 2 ] = origin[ 2 ];
378 
379  itk_image_->SetOrigin( point );
380 
381  typename image_type::DirectionType direction;
382 
383  direction[ 0 ][ 0 ] = direction_x[ 0 ];
384  direction[ 0 ][ 1 ] = direction_x[ 1 ];
385  direction[ 0 ][ 2 ] = direction_x[ 2 ];
386 
387  direction[ 1 ][ 0 ] = direction_y[ 0 ];
388  direction[ 1 ][ 1 ] = direction_y[ 1 ];
389  direction[ 1 ][ 2 ] = direction_y[ 2 ];
390 
391  direction[ 2 ][ 0 ] = direction_z[ 0 ];
392  direction[ 2 ][ 1 ] = direction_z[ 1 ];
393  direction[ 2 ][ 2 ] = direction_z[ 2 ];
394 
395  itk_image_->SetDirection( direction );
396 
397  typename image_type::SpacingType spacing;
398  spacing[ 0 ] = spacing_x;
399  spacing[ 1 ] = spacing_y;
400  spacing[ 2 ] = spacing_z;
401  itk_image_->SetSpacing( spacing );
402 }
